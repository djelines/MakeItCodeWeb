                    class _GameLogicState extends State {
                        String title = "Loading...";
                        String heading = "Loading...";
                        String instruction = "Loading...";
                        String codeContent = "";
                        int currentStep = 1; 
                        final _firstFieldController = TextEditingController();
                        final _secondFieldController = TextEditingController(); // Added the controller for the second field
                        final _lastFieldController = TextEditingController();
                        bool _isValid = false;
                        String firstFieldFromJson = "";
                        bool secondFieldFromJson = false ; 
                        String lastFieldFromJson = "";
                      
                        @override
                        void initState() {
                          super.initState();
                          currentStep = widget.currentStep; // Ensure currentStep is initialized correctly
                          loadContent(currentStep);
                        }
                      
                        // Function to load the JSON file and retrieve the data
                        Future loadContent(int step) async {
                          String jsonString = await rootBundle.loadString('assets/game/html.json');
                          List jsonData = json.decode(jsonString);
                          User? user = FirebaseAuth.instance.currentUser;
                      
                          var stepData = jsonData.firstWhere((element) => element["step"] == step);
                      
                          setState(() {
                            currentStep = step;
                            title = stepData["title"];
                            heading = stepData["description"]["heading"];
                            instruction = stepData["description"]["instruction"];
                            codeContent = stepData["code"];
                            firstFieldFromJson = stepData["firstField"]; 
                            secondFieldFromJson = stepData["secondField"]; 
                            lastFieldFromJson = stepData["lastField"]; 
                      
                            // Reset fields and button
                            _firstFieldController.clear();
                            _secondFieldController.clear(); // Reset the second field
                            _lastFieldController.clear();
                      
                            _isValid = false;  // Disable the button on every step change
                          });
                      
                          // Save the current player's level in Firebase
                          await FirebaseFirestore.instance
                            .collection('Users')
                            .doc(user?.uid)
                            .collection('Portfolio')
                            .doc('levelMap') // Key where the level is stored
                            .set({'currentStep': currentStep}, SetOptions(merge: true));
                        }
                      
                      Future saveUserData() async {
                        // Load the JSON file
                        String jsonString = await rootBundle.loadString('assets/game/html.json');
                        List jsonData = json.decode(jsonString);
                      
                        // Retrieve data for the current step
                        var stepData = jsonData.firstWhere((element) => element["step"] == currentStep);
                        String fieldKey = stepData["bdd"]; // Associated key (e.g., "html_structure", "footer_info", etc.)
                      
                        // Get the currently logged-in user
                        User? user = FirebaseAuth.instance.currentUser;
                      
                        if (user != null) {
                          try {
                            // Create an object to store the data
                            Map dataToSave = {};
                      
                            // Check that the key is not empty and associate the value entered by the user
                            if (fieldKey.isNotEmpty) {
                              dataToSave[fieldKey] = _secondFieldController.text.trim();
                            }
                      
                            // Save the data in Firebase
                            await FirebaseFirestore.instance
                                .collection('Users')
                                .doc(user.uid)
                                .collection('Portfolio')
                                .doc('data')
                                .set(dataToSave, SetOptions(merge: true)); // Merge to avoid overwriting other existing data
                      
                            print('Data saved in Firebase with UID: ${user.uid}');
                          } catch (e) {
                            print("Error saving data: $e");
                          }
                        } else {
                          print("No user logged in.");
                        }
                      }
                      
                        // Check if the fields entered by the user are correct
                        void _checkAnswer() {
                          setState(() {
                            _isValid = _firstFieldController.text.trim() == firstFieldFromJson &&
                                       _lastFieldController.text.trim() == lastFieldFromJson;
                          });
                        }
                      
                        void _nextStep() {
                          if (_isValid) {
                            // Trigger the function to load the next step
                            // Call the callback to increment the level in mapLevel
                            widget.onStepValidated(_isValid);  // Calls the callback function
                            saveUserData(); // Save data in Firebase
                            loadContent(currentStep + 1);
                      
                            if (currentStep == 20) {
                              Navigator.push(
                                context,
                                MaterialPageRoute(
                                  builder: (context) => Rewardscreen(
                                    xpToAdd: 500, // Example XP to add, you can customize it
                                  ),
                                ),
                            );
                          }
                          }
                        }

                    class GamePendu extends StatefulWidget {

                    class _GamePenduState extends State {
                      //List words to guess
                      final List words = [
                        "variable",
                        "fonction",
                        "classe",
                        "objet",
                        "boucle",
                        "tableau",
                        "condition",
                        "compilation",
                        "algorithme",
                        "debug"
                      ];
                      late String selectedWord;
                      late List guessedLetters;
                      late int attemptsLeft;
                    
                      @override
                      void initState() {
                        super.initState();
                        startNewGame();
                      }
                    
                    // function for starting a new game
                      void startNewGame() {
                        setState(() {
                          // Shuffle list and take firts word
                          words.shuffle();
                          selectedWord = words.first;
                          // Reset letter for new game
                          guessedLetters = [];
                          attemptsLeft = 6;
                        });
                      }
                    
                    // function for guessing a letter
                      void guessLetter(String letter) {
                        setState(() {
                          if (!guessedLetters.contains(letter)) {
                            guessedLetters.add(letter);
                            if (!selectedWord.contains(letter)) {
                              attemptsLeft--;
                            }
                          }
                        });
                      }
                    
                    // function for win the game
                      void _Victory() {
                        if (selectedWord
                            .split('')
                            .every((letter) => guessedLetters.contains(letter))) {
                          Navigator.push(
                            context,
                            MaterialPageRoute(
                              builder: (context) => RewardscreenXp(
                                  xpToAdd: 50,
                                  title:
                                      'Vous avez trouv√© le mot : $selectedWord !
                                      \n\nVous avez gagn√© 50 xp'),
                            ),
                          );
                        } else if (attemptsLeft == 0) {
                          print('Perdu !');
                        }
                      }

                      Widget build(BuildContext context) {
                        return Scaffold(
                          appBar: AppBar(
                            title: Text(

                                    fontWeight: FontWeight.bold,
